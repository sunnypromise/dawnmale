# craco对移动端适配

# 为什么要使用craco？

使用create-react-app脚手架创建的项目默认是不能修改内部webpack配置的，不像vue-cli那样通过配置文件(vue.config.js)修改，虽然script有个eject的命令可以使webpack配置暴露出来，但是这个是一个完全不可逆的操作同时也会失去CRA带来的便利和后续升级。

# 安装craco

 npm install  @craco/craco 
 or
 yarn add  @craco/craco
 当前版本："@craco/craco": "^7.0.0",

# 在根目录创建craco.config.js配置文件

  const path = require("path");
const pxtorem = require("postcss-pxtorem");
const pathResolve = (pathUrl) => path.join(__dirname, pathUrl); //配置别名

module.exports = {
  style: {
    postcss: {
      mode: "extends",
      loaderOptions: {
        postcssOptions: {
          ident: "postcss",
          plugins: [
            // [
            //   "postcss-px-to-viewport-8-plugin",
            //   {
            //     viewportWidth: 375, // 设计稿的视口宽度
            //   },
            // ],
            pxtorem({
              rootValue: 37.5,
              propWhiteList: [],
              minPixelValue: 2,
              exclude: /node_modules/i,
            }),
          ],
        },
      },
    },
  },
  webpack: {
    alias: {
      "@": pathResolve("src/")
    },
  },
  //   babel: {
  //     plugins: [
  //       ['import', { libraryName: 'antd', style: true }],
  //       ['@babel/plugin-proposal-decorators', { legacy: true }]
  //     ]
  //   },
  plugins: [],
};
