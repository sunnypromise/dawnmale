    },
  };

入口文件动态计算尺寸

  useEffect(() => {
    if (typeof window !== 'undefined') {
      // 动态计算html宽度
      const oHtml = document.getElementsByTagName('html')[0];
      const width = oHtml.clientWidth;
      const finalWidth = width > 720 ? 720 : width;
      oHtml.style.fontSize = `${20 * (finalWidth / 375)}px`; //最大宽度限制到720px
    }
  }, []);
