# next.config.js

const withTM = require('next-transpile-modules')([
  'antd-mobile',
]);

  webpack5: true,
  reactStrictMode: true, // 严格模式

      {
        source: '/',
        destination: '/home',
        permanent: true,
      },
    ];
  },

})

# 创建_app.tsx

import type { AppProps } from "next/app";
import { appWithTranslation } from "next-i18next";
import { ConfigProvider } from "antd-mobile";
import { useMemo } from "react";
import { useRouter } from "next/router";
import zhHK from "antd-mobile/es/locales/zh-HK";
import zhTW from "antd-mobile/es/locales/zh-TW";
import zhCN from "antd-mobile/es/locales/zh-CN";
function MyApp({ Component, pageProps }: AppProps) {
  const { locale } = useRouter();
  const usedLocale = useMemo(() => {
    switch (locale) {
      case "zh-HK":
        return zhHK;
      case "zh-TW":
        return zhTW;
      default:
        return zhCN;
    }
  }, [locale]);
  return (
    <>
      <ConfigProvider locale={usedLocale}>
        <Component {...pageProps} />
      </ConfigProvider>
    </>
  );
}

export default appWithTranslation(MyApp);

# 组件中使用

因为nextjs采用的是约定式路由，serverSideTranslations必須且只能在 pages 頁面層級之下才會生效 通过useTranslation实现语种的转化

import type { NextPage, GetServerSideProps } from "next";
import React from "react";
import { useTranslation } from "react-i18next";
import { serverSideTranslations } from "next-i18next/serverSideTranslations";
import { login } from "../../utils/helper";
const Home: NextPage = () => {
  const { t } = useTranslation("user");
  return (
    <div>
      {t("test")}
    </div>
  );
};
export const getServerSideProps: GetServerSideProps = async ({ locale }) => {
  return {
    props: {
      ...(await serverSideTranslations(locale || "", ["user"])),
    },
  };
};
export default Home;
