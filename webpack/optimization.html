# 压缩js体积

 npm i uglifyjs-webpack-plugin -D
 在vue.config.js进行配置
 const uglifyWebpackPlugin = require("uglifyjs-webpack-plugin");
     config.plugins.push(
      new uglifyWebpackPlugin({
        uglifyOptions: {
          uglifyOptions: {
            compress: {
              drop_debugger: true,
              drop_console: true, //生产环境删除console
            },
            warnings: false,
          },
          sourceMap: false,
          parallel: false, //使用多进程并行运行来提高构建速度
        },
      })
    );

# 图片压缩


npm install --save-dev image-webpack-loader(可能会有下载失败的情况可以选择用cnpm 下载)
 /* 对内部的 webpack 配置（比如修改、增加Loader选项）(链式操作) */
  chainWebpack: (config) => {
    // 压缩图片插件
    config.module
      .rule("images")
      /* Cannot find module 'imagemin-gifsicle' use cnpm install --save-dev image-webpack-loader */
      .use("image-webpack-loader")
      .loader("image-webpack-loader")
      .options({
        mozjpeg: { progressive: true, quality: 65 },
        optipng: { enabled: false },
        pngquant: { quality: [0.8, 0.9], speed: 4 },
        gifsicle: { interlaced: false },
        webp: { quality: 75 },
      });
  }
