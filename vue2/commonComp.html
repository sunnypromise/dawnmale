        <template>
            <div class="search">
                <div class="search-form">
                    <div
                        v-for="(item, index) in searchConfig"
                        :key="index"
                        class="search-item"
                    >
                            <p class="search-title">{{ `${item.label}:` }}</p>
                            <el-date-picker
                            v-if="item.type === 'daterange'"
                            v-model="searchModel[item.key]"
                            type="daterange"
                            value-format="yyyy-MM-dd"
                            format="yyyy-MM-dd"
                            range-separator="至"
                            start-placeholder="开始日期"
                            end-placeholder="结束日期"
                            >
                            </el-date-picker>

                            <el-date-picker
                            v-if="item.type === 'dateweek'"
                            v-model="searchModel[item.key]"
                            type="week"
                            @change="changeWeek(searchModel[item.key],item.key)"
                            value-format="yyyy-MM-dd"
                            :format="weekdayStr"
                            size="large"
                            :picker-options="{ firstDayOfWeek: 1 }"
                            placeholder="选择周"
                            >
                            </el-date-picker>

                            <el-date-picker
                            v-if="item.type === 'date'"
                            v-model="searchModel[item.key]"
                            :placeholder="item.placeholder"
                            type="date"
                            value-format="yyyy-MM-dd"
                            format="yyyy-MM-dd"
                            >
                            </el-date-picker>
                            <el-input
                            v-if="item.type === 'input'"
                            v-model="searchModel[item.key]"
                            size="medium"
                            :clearable="item.clearable || true"
                            :placeholder="item.placeholder || '请输入'"
                            :maxlength="item.maxlength"
                            ></el-input>
                            <el-select
                            v-if="item.type === 'select'"
                            v-model="searchModel[item.key]"
                            size="medium"
                            style="width: 100%"
                            :clearable="item.clearable || true"
                            :filterable="item.filterable || true"
                            :disabled="item.disabled || false"
                            :multiple="item.multiple || false"
                            :allow-create="item.allowCreate"
                            :placeholder="item.placeholder || '请选择'"
                            >
                            <el-option
                                v-for="(selectItem, selectIndex) in item.selectList"
                                :key="selectIndex"
                                :label="selectItem[item.listLabel]"
                                :value="selectItem[item.listValue]"
                            ></el-option>
                            </el-select>
                        </div>
                        </div>
                        <div class="search-btn">
                        <el-button
                            size="medium"
                            type="primary"
                            @click="search"
                            :style="{
                            background: '#4a7ffa',
                            color: '#FFFFFF',
                            'font-size': '14px',
                            }"
                            >查询</el-button
                        >
                        <el-button
                            size="medium"
                            plain
                            @click="reset"
                            class="reset-btn"
                            :style="{
                            color: '#0B0B0B',
                            'font-size': '14px',
                            }"
                            >重置</el-button
                        >
                        </div>
                        <!-- 其他的按钮需求通过插槽传入 -->
                        <slot name="searchBtn" :searchData="searchModel"></slot>
                </div>
            </template>
            <script>
            import dayjs from "dayjs";
            import weekday from "dayjs/plugin/weekday";
            import "dayjs/locale/zh-cn";
            export default {
            name: "commonSearch",
            props: {
                searchConfig: {
                // 搜索条件配置项
                type: Array,
                required: true,
                default() {
                    return [];
                },
                },
                searchModel: {
                // 搜索条件绑定值
                type: Object,
                required: true,
                default() {
                    return {};
                },
                },
            },
            data() {
                return {
                weekdayStr:''
                };
            },
            methods: {
                search() {
                this.$emit("search", this.searchModel);
                },
                reset() {
                Object.keys(this.searchModel).forEach((item) => {
                    this.searchModel[item] = "";
                });
                },
                changeWeek(dateStr,key){
                dayjs.extend(weekday);
                dayjs.locale("zh-cn");
                this.weekdayStr = dayjs(dateStr).startOf("week").format("YYYY-MM-DD") + '~'+dayjs(dateStr).endOf("week").format("YYYY-MM-DD");
                this.searchModel[key]= this.weekdayStr;
                }
            },
            }
            </script>

        <template>
            <div class="all-contianer">
                <CommonSearch
                :searchConfig="searchConfig"
                :searchModel="searchModel"
                :searchReset="searchReset"
                @search="getData(1)"
                >
                </CommonSearch>

                <template v-if="showConfig.isShowTable">
                <div class="table-content">
                    <!-- 批量操作 -->
                    <div class="multiple-operation">
                    <slot name="multiple-operation" :selectionData="selectionData"></slot>
                    </div>
                    <el-table
                    v-loading="loading"
                    v-bind="tableConfig"
                    :header-cell-style="headerCellStyle"
                    :data="tableData"
                    @selection-change="selectionChange"
                    class="table-conintar"
                    >
                    <!-- 可选框（多选） -->
                    <el-table-column
                        label="选择"
                        v-if="selection === 'multiple' && showConfig.showCheckBox"
                        type="selection"
                        align="center"
                        width="50"
                        :reserve-selection="rowKey ? true : false"
                        :selectable="selectable"
                    />
                    <!-- 可选框（单选） -->
                    <el-table-column
                        v-else-if="selection === 'single' && showConfig.showCheckBox"
                        align="center"
                        width="50"
                    >
                        <template v-slot:default="scope">
                        <el-radio
                            v-model="selectionRadio"
                            :label="scope.$index"
                            :disabled="selectable ? !selectable(scope.row) : false"
                            @change="selectionChangeSingle(scope.row)"
                        >
                        </el-radio>
                        </template>
                    </el-table-column>
                    <el-table-column
                        v-for="item in tableColumn"
                        :key="item.prop"
                        :prop="item.prop"
                        :sort-orders="tableConfig.sortOrders"
                        :label="item.label"
                        :sortable="item.sortable"
                        :width="item.width"
                        :align="item.align"
                    >
                        <template v-if="item.slot" v-slot:default="scope">
                        <slot
                            :name="item.prop"
                            :row="scope.row"
                            :index="scope.$index"
                        ></slot>
                        </template>
                        <template v-slot:default="scope" v-else>
                        <!-- 数据字典 -->
                        <!-- <div v-if="item.dictCode">
                            {{ scope.row[item.prop] | filterStatus(dict[item.dictCode]) }}
                        </div> -->
                        <!-- 扩展定制化需求 -->
                        <div v-if="item.format">
                            {{ item.format(scope.row) }}
                        </div>
                        <div v-else-if="item.actionClo" class="action-list">
                            <template v-for="opsColumn in item.actionClo">
                            <div
                                v-if="opsColumn.isShow(scope.row)"
                                :key="opsColumn.text"
                                :type="opsColumn.type"
                                :size="opsColumn.size"
                                :disabled="opsColumn.disabled(scope.row)"
                                @click="
                                opsColumn.handleClick
                                    ? opsColumn.handleClick(scope.row)
                                    : ''
                                "
                                class="action-item"
                                :style="{ color: opsColumn.color }"
                            >
                                <template v-if="opsColumn.type === 'icon'">
                                <i :class="opsColumn.iconClass" />
                                </template>
                                <template v-if="opsColumn.type === 'button'">
                                <el-button
                                    :size="opsColumn.size"
                                    :type="opsColumn.btnType"

            <script>
            import CommonSearch from "./serarch.vue";
            export default {
                name: "default-table",
                components: {
                    CommonSearch,
                },
                props: {
                    tableConfig: {
                        // 表格的基本配置属性
                        type: Object,
                        default() {
                            return {
                                stripe: false,
                                border: true,
                                showHeader: true,
                                sortOrders: ["ascending", "descending", null],
                            };
                        },
                    },
                    otherConfig: {
                                //接口响应字段配置
                                type: Object,
                                default: () => {
                                return {
                                 list: "list",
                            };
                        },
                    },
                    showConfig: {
                        // 显示配置项
                        type: Object,
                        default() {
                            return {
                                isShowTable: true,
                                showCheckBox: false,
                            };
                        },
                    },
                    searchConfig: {
                    // 搜索条件配置项
                        type: Array,
                        default: () => [],
                        },
                    searchReset: {
                        // 搜索条件默认值重置值
                        type: Object,
                    },
                    dict: {
                        // 字典配置
                        type: Object,
                        default: () => {},
                    },
                    apiUrl: {
                        // 列表接口（必填）
                        type: Function,
                        required: true,
                    },
                    tableColumn: {
                        // 自定义列配置
                        type: Array,
                        default: () => [],
                    },
                    selection: {
                        // 是否显示可选框（多选-multiple 、单选-single ）
                        type: String,
                        default: () => "multiple",
                    },
                    selectable: {
                        // 当前行是否可选择
                        type: Function,
                    },
                    rowKey: {
                        type: [Number, String, Function],
                        default: "",
                    },
                    pagination: {
                        // 是否需要分页，默认需要
                        type: Boolean,
                        default: true,
                    },
                    pageSizeArr: {
                        // 显示个数预设置
                        type: Array,
                        default: () => [10, 20, 50, 100],
                    },
                    emptyText: {
                        type: String,
                        default: () => "没有数据的展示文字",
                    },
                    headerCellStyle: {
                        // table-hader样式
                        type: Object,
                        default() {
                            return {
                            background: "#F5F7FA",
                            "border-radius": "4px 4px 0px 0px",
                            height: "50px",
                            };
                        },
                    },
                    paginationAlign: {
                        // 分页布局
                        type: String,
                        default: () => "left",
                    },
                },
                filters: {
                    // 字典项过滤
                    filterStatus(value, array) {
                    const findDictValue = array.filter((ite) => ite.code == value);
                    if (findDictValue) {
                        const [item] = findDictValue;
                        return item.name;
                    } else {
                        // 没有匹配的就原样返回
                        return value;
                    }
                    },
                },
                data() {
                    return {
                        searchModel: this.searchReset ? JSON.parse(JSON.stringify(this.searchReset)) : {},
                        tableData: [],
                        selectionData: [],
                        tableTotal: 0,
                        selectionRadio: "",
                        loading: true,
                        pageInfo: {
                            pageSize: 10,
                            pagesNo: 1,
                        },
                    };
                },
                created() {
                    this.getData();
                },
                methods: {
                    getData(startPage) {
                        if (startPage) {
                            this.pageInfo.pagesNo = startPage;
                        }
                    this.loading = true;
                    const requestFn = this.apiUrl;
                    const pageInfo = {
                        // 分页信息
                        pageSize: this.pageInfo.pageSize,
                        pagesNo: this.pageInfo.pagesNo,
                    };
                    let params = {
                        ...this.searchModel,
                    };
                    if (this.pagination) {
                        // 如果需要分页，则传分页信息
                        params = { ...params, ...pageInfo };
                    }
                    if (this.showConfig.isShowTable) {
                        requestFn().then((res) => {
                        // this.tableData = res.result.records || [];
                        this.tableData = res[this.otherConfig.list] || [];
                        this.tableTotal = res?.total || 0;
                        this.loading = false;
                        });
                    }
                    },
                    selectionChange(selection) {
                        this.selectionData = selection;
                    },
                    // 单选，选择行数据change
                    selectionChangeSingle(selection) {
                      this.selectionData = [selection];
                    },
                    sizeChange(size) {
                        this.pageInfo.pagesNo = 1;
                        this.pageInfo.pageSize = size;
                        this.getData();
                    },

                    // 页码变化
                    pageChange(page) {
                        this.pageInfo.pagesNo = page;
                        this.getData();
                    },
                },
            };
            </script>
