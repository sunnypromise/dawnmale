      v-report="{
        color: 'white',
        text: 'hello!',
        testFn: this.clickFn,
        triggerType: 'browse',
        testFn:this.clickFn
      }"
    >
      report
    </button>

# 在report内部

/* eslint-disable no-unused-vars */
import Axios from "axios";
const report = {
  bind(el, binding, vnode) {
    if (binding.value) {
      const { color, text, triggerType, testFn } = binding.value || {};
      testFn();
      let parmas = { color, text };
      if (triggerType == "browse") {
        // 页面初始化执行 例如采集pv
        Axios({ method: "post", url: "/browse/browse", data: parmas });
      } else if (triggerType == "click") {
        // 给埋点元素绑定事件上报
        el.addEventListener(
          "click",
          (e) => {
            Axios({ method: "post", url: "/click/click", data: parmas });
          },
          false
        );
      }
    }
  },
  inserted() {}, // 被绑定元素插入父节点时调用
  update() {}, //所在组件的 VNode 更新时调用，但是可能发生在其子 VNode更新之前。指令的值可能发生了改变，也可能没有。但是你可以通过比较更新前后的值来忽略不必要的模板更
  unbind() {}, // 只调用一次，指令与元素解绑时调用、
  componentUpdated() {}, // 绑定的所有vnode全部更新才会调用
};
export default report;
