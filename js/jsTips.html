# 获取所有的cookie转化为对象

  const getAllCookie= ()=>{
    let cookies =document.cookie;
    let allCookies = {};
    for(let i=0;i<cookies.length;i++){
        allCookies[cookies[i].split("=")[0].trim()]=cookies[i].split("=")[1]
    }
    return allCookies
  }

# 获取地址栏?后边的参数

        const changeUrl = () => {
        var query = location.search.substr(1)
        query = query.split('&')
        var params = {}
        for (let i = 0; i < query.length; i++) {
            let q = query[i].split('=')
            if (q.length === 2) {
                params[q[0]] = q[1]
            }
        }
        return params
    }
